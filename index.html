1. مقدمة عن JavaScript

JavaScript هي لغة برمجة تستخدم لإنشاء صفحات ويب تفاعلية. يمكن استخدامها لإضافة حركة وتفاعل لصفحات HTML.

نصيحة: JavaScript تعمل في المتصفح مباشرة، لا تحتاج لخادم!

كيفية استخدام JavaScript:

// يمكن كتابة JavaScript في 3 أماكن: // 1. داخل HTML مباشرة <script> console.log("مرحباً!"); </script> // 2. في ملف خارجي <script src="js.js"></script> // 3. في سطر واحد <button onclick="alert('مرحباً')">اضغط</button>

2. المتغيرات (Variables)

المتغيرات هي صناديق لحفظ البيانات. يمكنك تخزين أي نوع من البيانات فيها.

طرق تعريف المتغيرات:

// 1. var (الطريقة القديمة) var name = "أحمد"; // 2. let (يمكن تغييرها) let age = 25; age = 26; // ✅ يمكن التغيير // 3. const (ثابتة - لا يمكن تغييرها) const pi = 3.14; // pi = 3.15; ❌ خطأ! لا يمكن التغيير

مثال عملي:

let userName = "محمد"; let userAge = 30; console.log("الاسم: " + userName); console.log("العمر: " + userAge);

                        3. أنواع البيانات (Data Types)
                    

                    

                        
JavaScript تدعم أنواع مختلفة من البيانات:

                        
                        

                            
// 1. String (نص)
let name = "أحمد";
let message = 'مرحباً';

// 2. Number (رقم)
let age = 25;
let price = 99.99;

// 3. Boolean (صحيح/خطأ)
let isActive = true;
let isDeleted = false;

// 4. Array (مصفوفة)
let colors = ["أحمر", "أزرق", "أخضر"];

// 5. Object (كائن)
let person = {
    name: "أحمد",
    age: 25
};

// 6. null (فارغ)
let data = null;

// 7. undefined (غير معرف)
let x;
// x = undefined
                            
                        


                        

                            
                            نصيحة: استخدم typeof لمعرفة نوع البيانات: typeof name يعطي "string"

                        4. العوامل (Operators)
                    

                    

                        
العوامل الحسابية:

                        

                            
let a = 10;
let b = 5;

console.log(a + b); // 15 (جمع)
console.log(a - b); // 5 (طرح)
console.log(a * b); // 50 (ضرب)
console.log(a / b); // 2 (قسمة)
console.log(a % b); // 0 (باقي القسمة)
                            
                        


                        
عوامل المقارنة:

                        

                            
let x = 5;
let y = 10;

console.log(x == y); // false (يساوي)
console.log(x != y); // true (لا يساوي)
console.log(x < y); // true (أصغر من)
console.log(x > y); // false (أكبر من)
console.log(x <= y); // true (أصغر أو يساوي)
console.log(x >= y); // false (أكبر أو يساوي)
                            
                        

                    


                    
                    

                        
 اختبار: العوامل

                        5. الشروط (Conditions)
                    

                    

                        
الشروط تستخدم لاتخاذ قرارات في الكود بناءً على شروط معينة.

                        
                        
if / else if / else:

                        

                            
let age = 20;

if (age >= 18) {
    console.log("يمكنك القيادة");
} else if (age >= 16) {
    console.log("قريباً يمكنك القيادة");
} else {
    console.log("لا يمكنك القيادة بعد");
}
                            
                        


                        
Ternary Operator (عامل شرطي مختصر):

                        

                            
let age = 20;
let status = (age >= 18) ? "بالغ" : "قاصر";

// نفس الكود أعلاه لكن بطريقة مختصرة
// إذا كان الشرط صحيح: استخدم "بالغ"
// إذا كان خطأ: استخدم "قاصر"
                            
                        


                        
Switch:

                        

                            
let day = "السبت";

switch (day) {
    case "السبت":
        console.log("يوم إجازة!");
        break;
    case "الأحد":
        console.log("يوم عمل");
        break;
    default:
        console.log("يوم عادي");
}

                        6. الحلقات (Loops)
                    

                    

                        
الحلقات تستخدم لتكرار تنفيذ كود معين عدة مرات.

                        
                        
for Loop:

                        

                            
// طباعة الأرقام من 1 إلى 5
for (let i = 1; i <= 5; i++) {
    console.log(i);
}
// النتيجة: 1, 2, 3, 4, 5
                            
                        


                        
while Loop:

                        

                            
let i = 1;
while (i <= 5) {
    console.log(i);
    i++;
}
                            
                        


                        
forEach (للمصفوفات):

                        

                            
let colors = ["أحمر", "أزرق", "أخضر"];

colors.forEach(function(color) {
    console.log(color);
});
                            
                        

                    


                    
                    

                        
 اختبار: الحلقات

                        7. الدوال (Functions)
                    

                    

                        
الدوال هي كود قابل لإعادة الاستخدام. بدلاً من كتابة نفس الكود عدة مرات، نكتبه مرة واحدة في دالة.

                        
                        
طريقة تعريف الدالة:

                        

                            
// 1. Function Declaration
function greet(name) {
    return "مرحباً " + name;
}

// 2. Arrow Function (الطريقة الحديثة)
const greet2 = (name) => {
    return "مرحباً " + name;
};

// 3. Arrow Function مختصرة
const greet3 = (name) => "مرحباً " + name;

// استخدام الدالة
console.log(greet("أحمد")); // "مرحباً أحمد"
                            
                        


                        

                            
 مثال: دالة جمع رقمين

                            

                                
function add(a, b) {
    return a + b;
}

let result = add(5, 3);
console.log(result); // 8

                        8. المصفوفات (Arrays)
                    

                    

                        
المصفوفات هي قوائم لحفظ عدة قيم مرتبة.

                        
                        
إنشاء مصفوفة:

                        

                            
let fruits = ["تفاح", "موز", "برتقال"];

// الوصول للعناصر
console.log(fruits[0]); // "تفاح" (يبدأ من 0)
console.log(fruits.length); // 3 (عدد العناصر)
                            
                        


                        
طرق شائعة للمصفوفات:

                        

                            
let fruits = ["تفاح", "موز"];

// إضافة عنصر في النهاية
fruits.push("برتقال");

// حذف آخر عنصر
fruits.pop();

// إضافة عنصر في البداية
fruits.unshift("فراولة");

// حذف أول عنصر
fruits.shift();

// البحث عن عنصر
fruits.indexOf("موز"); // 1

                        9. الكائنات (Objects)
                    

                    

                        
الكائنات هي مجموعة من البيانات مرتبطة ببعضها.

    age: 25,
    city: "القاهرة",
    isActive: true
};

// الوصول للبيانات
console.log(person.name); // "أحمد"
console.log(person["age"]); // 25

// تعديل البيانات
person.age = 26;

// إضافة خاصية جديدة
person.email = "ahmed@example.com";

                        10. DOM (Document Object Model)
                    

                    

                        
DOM هو تمثيل HTML في JavaScript. يمكنك التلاعب بعناصر HTML.

                        
                        
اختيار العناصر:

                        

                            
// اختيار عنصر واحد
let title = document.querySelector("#title");
let button = document.querySelector(".btn");

// اختيار عدة عناصر
let allButtons = document.querySelectorAll(".btn");

// الطرق القديمة
let element = document.getElementById("myId");
let elements = document.getElementsByClassName("myClass");
                            
                        


                        
تغيير المحتوى:

                        

                            
let title = document.querySelector("#title");

// تغيير النص
title.textContent = "عنوان جديد";
title.innerHTML = "<strong>عنوان</strong>";

// تغيير الألوان
title.style.color = "red";
title.style.backgroundColor = "blue";

// إضافة/حذف class
title.classList.add("active");
title.classList.remove("active");
title.classList.toggle("active");

                        11. الأحداث (Events)
                    

                    

                        
الأحداث هي تفاعلات المستخدم مثل النقر، الكتابة، التمرير.

                        
                        
طريقة إضافة Event Listener:

                        

                            
let button = document.querySelector("#myButton");

// طريقة 1: addEventListener (الأفضل)
button.addEventListener("click", function() {
    alert("تم النقر!");
});

// طريقة 2: Arrow Function
button.addEventListener("click", () => {
    console.log("تم النقر!");
});

// طريقة 3: في HTML مباشرة
<button onclick="alert('تم النقر!')">اضغط</button>
                            
                        


                        
أحداث شائعة:

                        

                            
// click - النقر
button.addEventListener("click", function() {});

// mouseover - عند المرور بالماوس
button.addEventListener("mouseover", function() {});

// input - عند الكتابة في input
let input = document.querySelector("#myInput");
input.addEventListener("input", function(e) {
    console.log(e.target.value);
});

// submit - عند إرسال form
let form = document.querySelector("#myForm");
form.addEventListener("submit", function(e) {
    e.preventDefault(); // منع الإرسال الافتراضي
});

                        12. LocalStorage
                    

                    

                        
LocalStorage يستخدم لحفظ البيانات في المتصفح حتى بعد إغلاقه.

                        
                        

                            
// حفظ بيانات
localStorage.setItem("name", "أحمد");
localStorage.setItem("age", 25);

// قراءة بيانات
let name = localStorage.getItem("name");
let age = localStorage.getItem("age");

// حذف بيانات
localStorage.removeItem("name");

// حذف كل البيانات
localStorage.clear();

// حفظ كائن (يجب تحويله لـ JSON)
let user = { name: "أحمد", age: 25 };
localStorage.setItem("user", JSON.stringify(user));

// قراءة كائن
let savedUser = JSON.parse(localStorage.getItem("user"));
                            
                        


                        

                            
                            نصيحة: LocalStorage يحفظ البيانات كـ string فقط! استخدم JSON.stringify() و JSON.parse() للكائنات.
